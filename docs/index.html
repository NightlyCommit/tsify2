tsify2

tsify 2

Browserify plugin for TypeScript sources

Installation

 
npm install tsify2 --save-dev

Basic Usage

 
import * as Browserify from "browserify";
import Tsify from "tsify2";

const tsify = Tsify({
    // TypeScript compiler options
});

let browserify = Browserify()
    .add('main.ts')
    .plugin(tsify, {})
    .bundle()
    .pipe(process.stdout);
;
